## C

记忆化搜索

$dp[i][j][k]$表示当前第i个数字选数字j和为k有多少种情况

```cpp
const int MAXN = 55;
const int mod = 998244353;

int n, m, k;
ll dp[MAXN][MAXN][MAXN * MAXN];

ll dfs(int now, int num, int sum)
{
    ll &f = dp[now][num][sum];
    if(f != -1)
    {
        return f;
    }
    ll ans = 0;
    if(now == n)
    {
        if(sum <= k)
        {
            return f = 1;
        }
        else return f = 0;
    }
    
    for(int i = 1; i <= m; i++)
    {
        ans += dfs(now+1, i, sum + i);
        // cout << ans << endl;
        ans %= mod;
    }
    return f = ans;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    cin >> n >> m >> k;
    for(int i = 0; i < MAXN; i++)
    {
        for(int j = 0; j < MAXN; j++)
        {
            for(int k = 0; k < MAXN * MAXN; k++)
            {
                dp[i][j][k] = -1;
            }
        }
    }
    ll ans = dfs(0, 0, 0) % mod;
    cout << ans << endl;


    return 0;
}
```

## E

要避免浮点数运算

我们用$flg[i][j] = 1$表示以i和j为起点所连接的直线已经被计算过了

那么对于i和j确定的线，我们判断所有的点，如果点k也在线上，$flg[i][k] = flg[j][k] = 1$

直线用Ax + By = C表示，由y = kx + b通分而来，避免一切浮点数运算

```cpp
const int MAXN = 305;
const double eps = 1e-10;
ll x[MAXN], y[MAXN];
int vis[MAXN][MAXN];
int n, K;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> K;
    for (int i = 1; i <= n; i++)
    {
        cin >> x[i] >> y[i];
    }

    if (K == 1)
    {
        cout << "Infinity" << endl;
        return 0;
    }

    int ans = 0;

    for (int i = 1; i <= n; i++)
    {
        for (int j = i + 1; j <= n; j++)
        {
            if(vis[i][j]) continue;
            vis[i][j] = 1;
            vector<int> v;
            v.push_back(i);
            v.push_back(j);
            ll dx = x[i] - x[j], dy = y[i] - y[j];
            for(int id = 1; id <= n; id++)
            {
                if(vis[i][id]) continue;
                if(id == i || id == j) continue;
                ll dx2 = x[i] - x[id], dy2 = y[i] - y[id];
                if(dx * dy2 == dy * dx2)
                {
                    v.push_back(id);
                }
            }
            if(v.size() >= K)
            {
                ans++;
                for(int ii = 0; ii < v.size(); ii++)
                {
                    for(int jj = ii+1; jj < v.size(); jj++)
                    {
                        vis[v[ii]][v[jj]] = 1;
                        vis[v[jj]][v[ii]] = 1;
                    }
                }
            }
        }
    }
    cout << ans << endl;
    return 0;
}
```

## F

连通性dp

$dp[i][j][0/1]$到了第i列，已经删除了j条边，是否连通

详情：https://blog.csdn.net/m0_51979134/article/details/124275800

```cpp
const int MAXN = 3005;
ll dp[MAXN][MAXN][2];

int p;
int n;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n >> p;
    dp[1][0][1] = dp[1][1][0] = 1;

    for(int i = 2; i <= n; i++)
    {
        dp[i][0][1] = 1;
        for(int j = 1; j <= i; j++)
        {
            if(j >= 1) dp[i][j][1] = (dp[i-1][j][0] + 3 * dp[i-1][j-1][1] + dp[i-1][j][1]) % p;
            if(j >= 2) dp[i][j][0] = (dp[i-1][j-2][1] * 2 + dp[i-1][j-1][0]) % p;
        }
    }

    for(int i = 1; i < n; i++)
    {
        ll ans = dp[n][i][1];
        cout << ans % p << ' ';
    }
    cout << endl;
    return 0;
}
```

## G

反演可得
$$
\begin{gathered}
\sum_{i=1}^{n} \sum_{j=i+1}^{n} \operatorname{dis}(i, j) \times \sum_{d \mid \operatorname{gcd}\left(a_{P_{k}}\right)} \varphi(d) \\
\sum_{d} \varphi(d) \sum_{i=1}^{n} \sum_{j=i+1}^{n} d i s(i, j) \times\left[d \mid \operatorname{gcd}\left(a_{p_{k}}\right)\right]
\end{gathered}
$$
题目变成了枚举d，然后将d的倍数的点权的点提取出来，形成一个个连通块（森林），每一个连通块的贡献加起来就是当前d的贡献

```cpp
const int MAXN = 1e5 + 5;
const int mod = 998244353;

#define int long long

int a[MAXN];
int n;
int vis[MAXN];
vector<int> edge[MAXN], fac[MAXN];
int sz[MAXN];

ll res;

void dfs(int u, int fa, int d)
{
    vis[u] = 1;
    sz[u] = 1;
    for (int to : edge[u])
    {
        if (to == fa || a[to] % d)
            continue;
        dfs(to, u, d);
        sz[u] += sz[to];
    }
}

void dfs2(int u, int fa, int d, int size)
{
    vis[u] = 0;
    int lst = size - sz[u] + 1;
    res += lst;
    for (int v : edge[u])
    {
        if (v == fa || a[v] % d)
            continue;
        res = (res + lst * sz[v] % mod) % mod;
        lst += sz[v];
    }
    for (int v : edge[u])
    {
        if (v == fa || a[v] % d)
            continue;
        dfs2(v, u, d, size);
    }
}

int Phi(int x)
{
    int res = 1;
    for (int i = 2; i * i <= x; ++i)
    {
        if (x % i)
            continue;
        res *= (i - 1);
        x /= i;
        while (x % i == 0)
            res *= i, x /= i;
    }
    if (x > 1)
        res = res * (x - 1);
    return res;
}

signed main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
        int x = a[i];
        for (int j = 1; j * j <= x; j++)
        {
            if (x % j == 0)
            {
                int l = j, r = x / j;
                fac[l].push_back(i);
                if (l != r)
                {
                    fac[r].push_back(i);
                }
            }
        }
    }
    ll ans = 0;
    for (int i = 1; i < n; i++)
    {
        int x, y;
        cin >> x >> y;
        edge[x].push_back(y);
        edge[y].push_back(x);
    }

    for (int d = 1; d < MAXN; d++)
    {
        res = 0;
        for (int u : fac[d])
        {
            if (vis[u] == 0)
            {
                dfs(u, 0, d);
            }
        }
        for (int u : fac[d])
        {
            if (vis[u])
            {
                dfs2(u, 0, d, sz[u]);
            }
        }
        ans = (ans + Phi(d) * res % mod) % mod;
    }

    for (int i = 1; i <= n; ++i)
        ans = (ans - a[i] + mod) % mod;
    cout << ans << endl;
    return 0;
}
```

## Ex

维护区间l, r的max - min - r + l的值

因为是排列，这个式子一定大于等于0

原理就是固定r，我们看到这个区间最小值，已经最小值的个数即可

如果k==0，那么我们对于每一个r，询问1-r的式子最小值，如果最小值是0，那么统计式子最小值个数，就是r为右端点的合法个数，详见cf good subsegments

那么对于新加入的r，我们有一下操作：

1. 对于1 - r-1的区间，因为r多了1，这个区间的式子就要-1

2. 在1 - r-1的区间，之前的min有负的贡献，如果a[r]的值比之前某区间的最小值更小，我们要把这个区间之前的min加回去，相当于强控，max类似，这个可以用单调栈维护

   >单调栈可以维护一个值是哪个极大区间的min和max，对于一个单调递增栈，如果一个元素进来弹出了其他元素，停下的位置就是这个值作为区间min极大区间的左端点。

3. 2之后，我们找到a[r]最为min和max的极大区间的左端点，加上作为max的贡献，减去作为min的贡献

k不等于0的时候，我们记录这个区间的式子的最小值+1， +2， +3的次数即可

以上用线段树维护

```cpp
const int MAXN = 2e5+5;
#define lch (rt << 1)
#define rch (rt << 1 | 1)

int n, k;

struct node
{
    int val;
    int l, r;
    int add, cnt[4];
}tree[MAXN << 2];

const int INF = 0x3f3f3f3f;

int a[MAXN];

void pushup(int rt)
{
    tree[rt].val = min(tree[lch].val, tree[rch].val);
    for(int i = 0; i <= k; i++)
    {
        tree[rt].cnt[i] = 0;
        int tmp = tree[rt].val - tree[lch].val + i;
        if(tmp >= 0)
        {
            tree[rt].cnt[i] += tree[lch].cnt[tmp];
        }
        tmp = tree[rt].val - tree[rch].val + i;
        if(tmp >= 0)
        {
            tree[rt].cnt[i] += tree[rch].cnt[tmp];
        }
    }
}

void pushdown(int rt)
{
    if(tree[rt].add)
    {
        tree[lch].add += tree[rt].add;
        tree[rch].add += tree[rt].add;
        tree[lch].val += tree[rt].add;
        tree[rch].val += tree[rt].add;
        tree[rt].add = 0;
    }
}

void build(int rt, int l, int r)
{
    tree[rt].l = l, tree[rt].r = r;
    if(l == r)
    {
        tree[rt].val = INF;
        tree[rt].cnt[0] = 1;
        return ;
    }
    int mid = l + r >> 1;
    build(lch, l, mid);
    build(rch, mid+1, r);
}

void modify(int ql, int qr, int rt, int l, int r, int v)
{
    if(ql > qr) return ;
    if(ql <= l && r <= qr)
    {
        tree[rt].add += v;
        tree[rt].val += v;
        return ;
    }
    pushdown(rt);
    int mid = l + r >> 1;
    if(ql <= mid)
    {
        modify(ql, qr, lch, l, mid, v);
    }
    if(mid < qr)
    {
        modify(ql, qr, rch, mid+1, r, v);
    }
    pushup(rt);
    return ;
}

int ask(int x, int rt, int l, int r)
{
    if(l ==x  && r == x)
    {
        return tree[rt].val;
    }
    pushdown(rt);
    int mid = l + r >> 1;
    if(x <= mid)
    {
        return ask(x, lch, l, mid);
    }
    else
    {
        return ask(x, rch, mid+1, r);
    }
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n >> k;
    for(int i = 1; i <= n; i++)
    {
        cin >> a[i];
    }
    build(1, 1, n);
    stack<pair<P, int>> smin, smax;
    ll ans = 0;
    for(int i = 1; i <= n; i++)
    {
        modify(i, i, 1, 1, n, -INF);
        modify(1, i-1, 1, 1, n, -1);
        int x1 = i, x2 = i;
        while(smin.size())
        {
            if(a[i] < smin.top().second)
            {
                modify(smin.top().first.first, smin.top().first.second, 1, 1, n, smin.top().second);
                x1 = smin.top().first.first;
                smin.pop();
            }
            else break;
        }
        while(smax.size())
        {
            if(a[i] > smax.top().second)
            {
                modify(smax.top().first.first, smax.top().first.second, 1, 1, n, -smax.top().second);
                x2 = smax.top().first.first;
                smax.pop();
            }
            else break;
        }
        smin.push({{x1, i}, a[i]});
        smax.push({{x2, i}, a[i]});
        modify(x1, i, 1, 1, n, -a[i]);
        modify(x2, i, 1, 1, n, a[i]);
        ll tans = 0;
        for(int c = 0; c <= k; c++)
        {
            if(tree[1].val + c <= k)
            {
                tans += tree[1].cnt[c];
            }
        }
        ans += tans;
    }
    cout << ans << endl;
    return 0;
}
```
